    <script>
        let map = null;
        let markers = [];
        
        function initMap() {
            map = new google.maps. Map(document.getElementById("map"), {
                zoom: 5,
                center:  { lat: 22.5, lng: 78.5 },
                styles: [{ elementType: "geometry", stylers: [{ color:  "#242f3e" }] }]
            });
            refreshData();
            setInterval(refreshData, 30000);
        }
        
        async function refreshData() {
            try {
                const statsRes = await fetch('/api/statistics');
                const stats = await statsRes. json();
                document.getElementById('fire-count').textContent = stats.fire_alerts || 0;
                document.getElementById('smoke-count').textContent = stats.smoke_alerts || 0;
                document. getElementById('critical-count').textContent = stats.critical_alerts || 0;
                document.getElementById('device-count').textContent = stats.active_devices || 0;
                
                const alertsRes = await fetch('/api/alerts');
                const data = await alertsRes.json();
                displayAlerts(data. alerts || []);
                updateMarkers(data.alerts || []);
            } catch (e) {
                console.error(e);
            }
        }
        
        function displayAlerts(alerts) {
            const container = document.getElementById('alerts-list');
            if (! alerts. length) {
                container.innerHTML = 'No alerts yet';
                return;
            }
            container.innerHTML = alerts.slice(0, 10).map(a => 
                '<div class="alert-item"><b>' + (a.fire_detected ? '🔥 Fire' : '💨 Smoke') + '</b> - ' + 
                a.risk_level. toUpperCase() + '<br>📍 ' + a.location_name + '</div>'
            ).join('');
        }
        
        function updateMarkers(alerts) {
            if (!map) return;
            markers. forEach(m => m.setMap(null));
            markers = [];
            
            const colors = { critical: "#c0392b", high: "#e74c3c", medium:  "#f39c12", low: "#27ae60" };
            
            alerts. filter(a => a. latitude && a.longitude).forEach(a => {
                const marker = new google.maps.Marker({
                    position: { lat: a.latitude, lng: a. longitude },
                    map: map,
                    icon: {
                        path: google.maps.SymbolPath.CIRCLE,
                        fillColor: colors[a.risk_level] || "#f39c12",
                        fillOpacity: 0.9,
                        strokeColor: "#fff",
                        strokeWeight: 2,
                        scale:  12
                    }
                });
                markers.push(marker);
            });
        }
        
        async function createTestAlert(risk) {
            await fetch('/api/test-alert', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ risk_level: risk })
            });
            refreshData();
        }
        
        async function clearAlerts() {
            await fetch('/api/clear-alerts', { method:  'POST' });
            refreshData();
        }
        
        window.onload = function() {
            setTimeout(function() {
                if (! map) {
                    document. getElementById('map').innerHTML = '<div style="text-align: center;padding:150px;color:#888;">🗺️ Map requires API key</div>';
                    refreshData();
                    setInterval(refreshData, 30000);
                }
            }, 3000);
        };
    </script>
